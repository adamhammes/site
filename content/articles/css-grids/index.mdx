```html
  </ul>

```

```scss
.css-grids--flex-outer {
  padding: 0 calc(0.5 * var(--grid-spacing));
}

.css-grids--flex {
  @include list-reset();
  display: flex;
  flex-wrap: wrap;
  margin: calc(-0.5 * var(--grid-spacing));
}

.css-grids--flex li {
  padding: calc(0.5 * var(--grid-spacing));

  @media screen and (min-width: 550px) {
    width: calc(100% / 2);
  }

  @media screen and (min-width: 800px) {
    width: calc(100% / 3);
  }
}
```
